 03: typedef struct node { 
 04:   int x; 
 05:   int y; 
 06:   struct node *next;

 11:   node_t *tmp;

 14:

 16:     tmp       = (node_t *) malloc(sizeof(node_t)); 
 17:     tmp->x    = i; 
 18:     tmp->y    = i+1; 
 19:     tmp->next = root; 
 20:     root      = tmp; 
 21:   } 
 22:  
 23:   for(tmp = root; tmp!=(node_t*)0; tmp = tmp->next){ 
 24:     assert(tmp->x >= 0);

 27:   }

 29: }
